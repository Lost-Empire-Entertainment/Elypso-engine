
----------------------------------------------------------------------
IMPORTANT LINUX SETUP COMMANDS
----------------------------------------------------------------------

# install gcc, g++, make and other essential build tools
sudo apt install build-essential -y

# install gnu debugger
sudo apt install gdb -y

# install remote gnu debugger
sudo apt install gdbserver -y

# install cmake
sudo apt install cmake -y

# install zenity
sudo apt install zenity -y

# install xdg-utils
sudo apt install xdg-utils -y

# install mesa drivers
sudo apt install mesa-vulkan-drivers mesa-utils -y

# install vulkan drivers
sudo apt install vulkan-tools vulkan-validationlayers -y

# install amd drivers (if using amd gpu and drivers arent installed)
sudo apt install mesa-vulkan-drivers mesa-opencl-icd libgl1-mesa-dri xserver-xorg-video-amdgpu -y

# look for latest nvidia drivers (if using nvidia gpu and nvidia drivers arent installed)
sudo add-apt-repository ppa:graphics-drivers/ppa
sudo apt update
apt search nvidia-driver

# install nvidia drivers (if using nvidia gpu and nvidia drivers arent installed, replace xxx with nvidia driver version)
sudo apt install nvidia-driver-xxx nvidia-utils-xxx -y

# install x11 drivers
sudo apt install libx11-dev libxrandr-dev libxinerama-dev libxcursor-dev -y

# install linux-compatible glfw3
sudo apt install libglfw3-dev -y

# install linux-compatible assimp
sudo apt install libassimp-dev -y

----------------------------------------------------------------------
COMPILE AND RUN ELYPSO ENGINE OR OTHER PROGRAM
----------------------------------------------------------------------

# configure with cmake and compile with make
username=$USER
cd /home/$username/.vs/
rm -rf build
cmake -S . -B build -DCMAKE_BUILD_TYPE=Release
cd build
make -j

# run Elypso engine in current folder
"./Elypso Engine"
